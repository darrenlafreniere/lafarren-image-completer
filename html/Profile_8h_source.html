00143 //
00144 // Macros for scoped time profiling. Use as such:
00145 //
00146 //              void MyClass::MyMethod()
00147 //              {
00148 //                      TECH_TIME_PROFILE("MyClass::MyMethod");
00149 //                      ...
00150 //
00151 // The time profiler macros create static TimeProfiler objects. Therefore,
00152 // the name passed into time profiler macros cannot change; the first name
00153 // is used throughout execution.
00154 //
00155 // The TECH_PROFILE preprocessor flag must be set to 1
00156 
00157 #if TECH_PROFILE_MACROS && defined(_MSC_VER)
00158 
00159 // Reports the final profile:
00160 #define TECH_TIME_PROFILE(__name__) \
00161         TECH_TIME_PROFILE_INDIRECT1(__LINE__, __name__, LfnTech::TimeProfiler::ReportFinal, 0)
00162 
00163 // Reports every sample:
00164 #define TECH_TIME_PROFILE_EVERY_SAMPLE(__name__) \
00165         TECH_TIME_PROFILE_INDIRECT1(__LINE__, __name__, LfnTech::TimeProfiler::ReportEverySample, 1)
00166 
00167 // Reports every Nth sample:
00168 #define TECH_TIME_PROFILE_EVERY_NTH_SAMPLE(__name__, __nthBlock__) \
00169         TECH_TIME_PROFILE_INDIRECT1(__LINE__, __name__, LfnTech::TimeProfiler::ReportEveryNthSample, __nthBlock__)
00170 
00171 // Helper macros to successfully expand __LINE__
00172 #define TECH_TIME_PROFILE_INDIRECT1(__line__, __name__, __reportMode__, __nthBlock__) \
00173         TECH_TIME_PROFILE_INDIRECT2(__line__, __name__, __reportMode__, __nthBlock__)
00174 
00175 #define TECH_TIME_PROFILE_INDIRECT2(__line__, __name__, __reportMode__, __nthBlock__) \
00176         static LfnTech::TimeProfiler techTimeProfiler##__line__(__name__, __reportMode__, __nthBlock__); \
00177         LfnTech::ScopedProfiler techScopedProfiler##__line__(techTimeProfiler##__line__)
00178 
00179 #else
00180 
00181 #define TECH_TIME_PROFILE(__name__)
00182 #define TECH_TIME_PROFILE_EVERY_SAMPLE(__name__)
00183 #define TECH_TIME_PROFILE_EVERY_NTH_SAMPLE(__name__, __nthBlock__)
00184 
00185 #endif // #if TECH_PROFILE_MACROS
