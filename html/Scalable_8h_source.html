00022 #ifndef SCALABLE_H
00023 #define SCALABLE_H
00024 
00025 namespace LfnIc
00026 {
00027         //
00028         // Interface for scaling an object up and down in-place. Implementators
00029         // of this interface must preserve data such that:
00030         //
00031         //      dataBefore = objectThatImplementsScalable.CloneData()
00032         //      objectThatImplementsScalable.ScaleDown()
00033         //      objectThatImplementsScalable.ScaleUp() // object was not mutated between scaling calls
00034         //      dataAfter = objectThatImplementsScalable.CloneData()
00035         //      assert(dataBefore == dataAfter)
00036         //
00037         class Scalable
00038         {
00039         public:
00040                 // Scales the instance's data up to the next higher resolution/depth.
00041                 // Asserts that this instance isn't at its original depth.
00042                 virtual void ScaleUp() = 0;
00043 
00044                 // Scales the instance's data down to the next lower resolution/depth.
00045                 virtual void ScaleDown() = 0;
00046 
00047                 // Returns the instance's current scale depth. 0 is the original
00048                 // resolution's depth, and 1 is the next lower resolution/depth.
00049                 virtual int GetScaleDepth() const = 0;
00050 
00051         protected:
00052                 // Cannot destroy Scalable instance through a pointer to this base.
00053                 ~Scalable() {}
00054         };
00055 
00056         //
00057         // Scales down each added Scalable instance, and when this goes out of
00058         // scope, scales them back up in the same order as they were added.
00059         //
00060         class ScopedScaleDownAndUpInOrder
00061         {
00062         public:
00063                 inline void Add(Scalable& scalable)
00064                 {
00065                         scalable.ScaleDown();
00066                         m_scalables.push_back(&scalable);
00067                 }
00068 
00069                 inline ~ScopedScaleDownAndUpInOrder()
00070                 {
00071                         for (int i = 0, n = m_scalables.size(); i < n; ++i)
00072                         {
00073                                 m_scalables[i]->ScaleUp();
00074                         }
00075                 }
00076 
00077         private:
00078                 std::vector<Scalable*> m_scalables;
00079         };
00080 }
